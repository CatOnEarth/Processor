408long long HashFunc(void* start_hash, void* end_hash) {
409 if (_IsBadReadPtr(start_hash)) {
410 StackAbort(nullptr, TypeError::_ERROR_SEGMENTATION_FAULT DEBUG_CODE_ADD(, LOCATION{ __FILE__, __FUNCTION__, __LINE__,
411 "void*", "start_hash" }));
412 }
413 if (_IsBadReadPtr(end_hash)) {
414 StackAbort(nullptr, TypeError::_ERROR_SEGMENTATION_FAULT DEBUG_CODE_ADD(, LOCATION{ __FILE__, __func__, __LINE__,
415 "void*", "end_hash" }));
416 }
417
418 if (!(start_hash < end_hash)) {
419 StackAbort(nullptr, TypeError::_ERROR_SEGMENTATION_FAULT DEBUG_CODE_ADD(, LOCATION{ __FILE__, __FUNCTION__, __LINE__,
420 "void*", "start_hash, end_hash" }));
421 }
422
423 long long res_hash_func = 0;
424 for (size_t num_byte = 0; num_byte < (char*)end_hash - (char*)start_hash; ++num_byte) {
425 if (_IsBadReadPtr((char*)start_hash + num_byte)) {
426 StackAbort(nullptr, TypeError::_ERROR_SEGMENTATION_FAULT DEBUG_CODE_ADD(, LOCATION{ __FILE__, __FUNCTION__, __LINE__,
427 "void*", "start_hash" }));
428 }
429 char byte_value = *((char*)start_hash + num_byte);
430 res_hash_func = ((res_hash_func << 5) + res_hash_func) + byte_value;
431 }
432
433 return res_hash_func;
434}
